## API Route Handler

```ts
// app/api/contact/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { Resend } from 'resend';

export async function POST(req: NextRequest) {
  const apiKey = process.env.RESEND_API_KEY;
  if (!apiKey) {
    return NextResponse.json({ error: 'Email service not configured' }, { status: 500 });
  }

  const { name, email, message } = await req.json();

  // Basic validation
  if (!name || !email || !message) {
    return NextResponse.json({ error: 'Missing required fields' }, { status: 400 });
  }

  const resend = new Resend(apiKey);
  const to = process.env.CONTACT_EMAIL ?? 'nicholaswilliam2307@gmail.com';

  const { error } = await resend.emails.send({
    from: 'Portfolio Contact <onboarding@resend.dev>',
    to,
    subject: `Portfolio message from ${name}`,
    text: `From: ${name} <${email}>\n\n${message}`,
  });

  if (error) {
    return NextResponse.json({ error: 'Failed to send message' }, { status: 500 });
  }

  return NextResponse.json({ success: true });
}
```

**Note on `from` address:** Resend requires a verified domain for custom from addresses. Until the domain is verified, use `onboarding@resend.dev` (Resend's shared domain) which works on free tier. The user's email appears in the email body and reply-to.

## ContactForm component

`'use client'` component. State: `status: 'idle' | 'loading' | 'success' | 'error'`.

```
[form onSubmit]
  [field: name] <input type="text" required />
  [field: email] <input type="email" required />
  [field: message] <textarea rows={5} required />
  <button type="submit" disabled={status === 'loading'}>
    {status === 'loading' ? 'Sending...' : 'Send message'}
  </button>

  {status === 'success' && <p class="success-msg">Message sent — I'll be in touch soon.</p>}
  {status === 'error' && <p class="error-msg">Something went wrong. Please try again.</p>}
```

Style conventions:
- Input/textarea: `w-full bg-[#1e1e1e] border border-white/10 rounded-md px-4 py-3 text-sm text-[#f5f5f5] placeholder:text-[#525252] focus:outline-none focus:ring-1 focus:ring-[#c8964a]`
- Label: `text-xs font-mono text-[#a3a3a3] tracking-widest uppercase mb-2 block`
- Submit button: use existing `<Button>` component (`components/ui/Button.tsx`) if it accepts `disabled` prop; otherwise style inline
- Success: `text-sm text-green-400 mt-4`
- Error: `text-sm text-red-400 mt-4`
